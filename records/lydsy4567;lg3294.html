输入格式

输入一个整数 n，表示 Lweb 要学习的单词数。

接下来 n 行，每行有一个单词(由小写字母构成，且保证任意单词两两互不相同)。

1 ≤ n ≤ 100000，所有字符的长度总和满足 。

输出格式

Lweb 吃的最少泡椒数。

题解

题解就一句话：Trie 加 3 个 dfs (有些人可能 2 个就可以啦)。下面详细讲解一下：

Trie 这个辣鸡玩意儿是来处理前缀的。那后缀怎么办呢？reverse 大发好。

如果你是 Lweb，那么你肯定不喜欢第 1 种，因为要吃 n2 个泡椒啊，听着都可怕……

(scx: 所以我制定的第 1 种计划就是来搞笑的(滑稽))

因此，就是说对于一个字符串，它的后缀不能出现在它后面，辣么 reverse 后它的前缀就不能出现在它的前面。

构造一棵 Trie，把没什么卵用的中间节点去掉，Tarjan缩点一下，就变成了一棵大树。

然后，这题就变成了给树上每个点赋予一个权值，使孩子节点的编号大于其父亲节点的编号。最小化 。

然后直觉告诉我们 dfs 序是最好的。

辣么一来，把子树大小排个序，先 dfs 子树较小的，然后计算一下就可以了。(这次的题解怎么那么短)

代码

#include <bits/stdc++.h>
#define N 100034
#define Sz 682936
using namespace std;

typedef long long ll;
typedef vector <int> vec;

int n, i, l;
char s[Sz];
ll ans;
// dfa
int cnt, d[Sz][27], val[Sz];
// dfs
int tim, dfn[Sz];
int size[N];
vec Vn[N];

inline bool cx(int A, int B){return size[A] < size[B];}

void insert(char *_bg, char *_en){
    int x = 0, ch;
    char *p;
    for(p = _bg; p < _en; ++p){
        ch = *p - 'a';
        x = (d[x][ch] ? d[x][ch] : d[x][ch] = ++cnt);
    }
    ++val[x];
}

void dfs_cpr(int x){
    if(val[x]){    
        Vn[dfn[x]].push_back(++tim);
        dfn[x] = tim;
    }
    for(int i = 0; i < 26; i++)
        if(d[x][i]){
            d[x][i][dfn] = dfn[x];
            dfs_cpr(d[x][i]);
        }
}

void dfs_size(int x){
    vec :: iterator it;
    size[x] = 1;
    for(it = Vn[x].begin(); it != Vn[x].end(); ++it){
        dfs_size(*it);
        size[x] += size[*it];
    }
    sort(Vn[x].begin(), Vn[x].end(), cx);
}

void dfs_final(int x){
    vec :: iterator it;
    dfn[x] = ++tim;
    for(it = Vn[x].begin(); it != Vn[x].end(); ++it){
        ans += (ll)(tim - dfn[x] + 1);
        dfs_final(*it);
    }
}

int main(){
    scanf("%d", &n);
    memset(d, 0, sizeof d);
    memset(val, 0, sizeof val);
    cnt = 0;
    for(i = 0; i < n; i++){
        scanf("%s", s);
        l = strlen(s);
        reverse(s, s + l);
        insert(s, s + l);
    }
    dfn[0] = tim = 0;
    dfs_cpr(0);
    dfs_size(0);
    ans = tim = 0;
    dfs_final(0);
    printf("%lld\n", ans);
    return 0;
}
